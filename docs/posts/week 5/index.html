avatar <- tibble(
  name = c("Katara", "Toph", "Sokka"),
  bends = c("water", "earth", NA),
  friendly = TRUE,
  numbers = c(1,2,3),
  more_numbers = 1:3,
  even_more = seq(1,3,1)
)
